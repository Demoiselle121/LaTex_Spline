% Leslie Lamport hat LaTeX entwickelt, um dem Anwender das Schreiben

% Vorstellungen eines Layouts zu entsprechen, hat Markus Kohm analoge
% Dokumentklassen entwickelt (Koma-Klassen)

% im Original    extern              Koma       extern

% 2. report                          scrreprt

% 5. slide                                      beamer

% Kopf des Dokumentes
% ===================
\documentclass[                 % neue Dokumentklasse, es gibt Alternativen,
                                % beamer hat sich als Standard durchgesetzt
                                % Optionen sind
                 % aspectratio= % Verhältnis der Seiten, muss auf den jeweiligen
                                % Projektor angepasst werden, möglichkeiten sind
                                % 169 (16:9), 1610 (16:10), ..., der Standard
                                % ist 4:3
                 handout,       % das Poster wird so formatiert,
                                % dass es auf einer Seite erscheint
                 % ignorenonframetext,
                 ngerman,       % die üblichen globalen Optionen werden
                                % unterstützt
            ]{beamer}
\usepackage[orientation=portrait, size=a0, scale=1.4]{beamerposter}
%           ^                     ^        ^
%           |                     |        Die Schriftgröße wird um
%           |                     |        den Faktor 1.4 vergrößert
%           |                     Die Zielgröße ist DIN A4
%           Das Zielformat ist Hochformat
%           im Gegensatz zu Querformat (landscape)
